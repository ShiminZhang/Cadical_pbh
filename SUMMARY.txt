âš™ï¸ CADICAL OPTIONS
==================

Set options using:
  -o <name>=<value>
  --<name>=<value>
  --<name>           (for boolean, set to true)
  --no-<name>        (for boolean, set to false)

Common Options:
  --check=1          Enable internal proof checking
  --chrono=0         Disable chronological backtracking
  --phase=1          Initial phase (0=false, 1=true)
  --stable=0         Disable stable mode
  --verbose=1        CaDiCaL verbose output

Examples:
  $ ./priority_branch -o check=1 problem.cnf branches.txt
  $ ./priority_branch --no-stable problem.cnf branches.txt
  $ ./priority_branch -o phase=0 -o chrono=0 problem.cnf branches.txt

ğŸ’» IN YOUR C++ CODE
===================

#include "src/cadical.hpp"
#include "src/priority_branching_propagator.hpp"

int main() {
    CaDiCaL::Solver solver;
    CaDiCaL::PriorityBranchingPropagator propagator(false);
    
    // Enable proof BEFORE reading CNF (important!)
    solver.trace_proof("proof.drat");
    
    // Set options
    solver.set("check", 1);
    solver.set("verbose", 0);
    
    // Read CNF
    int vars = 0;
    solver.read_dimacs("problem.cnf", vars, 1);
    
    // Connect propagator
    solver.connect_external_propagator(&propagator);
    
    // Load branch list
    propagator.load_branch_list_from_file("branches.txt");
    
    // Mark observed variables (REQUIRED!)
    for (int lit : propagator.get_branch_on_list()) {
        solver.add_observed_var(abs(lit));
    }
    
    // Solve
    int result = solver.solve();
    
    // Flush and close proof
    solver.flush_proof_trace();
    solver.close_proof_trace();
    
    // Clean up
    solver.disconnect_external_propagator();
    
    return result == 10 ? 0 : 1;
}
